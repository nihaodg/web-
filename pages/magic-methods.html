PHP 魔术方法详解

什么是魔术方法？

PHP魔术方法是以__（双下划线）开头的特殊方法，在特定场景下自动触发。 它们是反序列化漏洞利用的核心，我们需要理解每个方法的触发条件和可利用的操作。

构造与析构方法

__construct()

对象创建时触发

对象被创建时自动调用，常用于初始化对象属性。

PHP代码 

class User {
    public $name;
    public $age;
    
    public function __construct($name, $age) {
        $this->name = $name;
        $this->age = $age;
        echo "构造函数被调用: " . $this->name . "\n";
    }
}

$user = new User("Alice", 20);
运行结果 
构造函数被调用: Alice

__destruct()

对象销毁时触发

对象被销毁时自动调用（脚本结束、unset()、或引用计数归零）。反序列化漏洞中最常用的触发点！

PHP代码 

class Demo {
    public $data;
    
    public function __destruct() {
        echo "对象被销毁，最后操作: " . $this->data . "\n";
    }
}

// 当脚本结束时，未使用的对象会被销毁
$demo = new Demo();
$demo->data = "cleanup";
// 脚本结束，对象销毁
运行结果 
对象被销毁，最后操作: cleanup

属性访问方法

__get($name)

读取不可访问属性时触发

当读取一个不存在的属性或protected/private属性时自动调用。可用于日志记录或动态计算属性值。

PHP代码 

class User {
    private $$secret = "password123";
    
    // 当访问不存在或不可访问的属性时调用
    public function __get($name) {
        echo "尝试获取属性: $name\n";
        return "你访问了" . $name;
    }
}

$user = new User();
echo $user->nonExistent . "\n";
echo $user->name;
运行结果 
尝试获取属性: nonExistent
你访问了nonExistent
尝试获取属性: name
你访问了name

__set($name, $value)

修改不可访问属性时触发

当对不存在的属性或protected/private属性进行赋值时自动调用。

PHP代码 

class User {
    private $$data = [];
    
    public function __set($name, $value) {
        echo "设置属性 $name = $value\n";
        $this->$data[$name] = $value;
    }
}

$user = User();
$user->email = "alice@example.com";
$user->role = "admin";
运行结果 
设置属性 email = alice@example.com
设置属性 role = admin

__isset($name)

使用isset()/empty()时触发

当对不可访问的属性使用isset()或empty()时自动调用。

PHP代码 

class Config {
    private $$settings = ["debug" => true];
    
    public function __isset($name) {
        echo "检查属性是否存在: $name\n";
        return isset($this->$settings[$name]);
    }
}

$config = new Config();
var_dump(isset($config->debug));
var_dump(isset($config->nonexistent));
运行结果 
检查属性是否存在: debug
bool(true)
检查属性是否存在: nonexistent
bool(false)

__unset($name)

使用unset()时触发

当对不可访问的属性使用unset()时自动调用。

PHP代码 

class Safe {
    private $$data = ["secret" => "hidden"];
    
    public function __unset($name) {
        echo "尝试删除属性: $name\n";
        if (isset($this->$data[$name])) {
            unset($this->$data[$name]);
            echo "已删除\n";
        }
    }
}

$safe = new Safe();
unset($safe->secret);
运行结果 
尝试删除属性: secret
已删除

字符串操作方法

__toString()

对象被当作字符串时触发 漏洞利用核心

当对象被当作字符串使用时触发，如： echo $obj、 print $obj、 字符串拼接 $obj . "str"、 字符串格式化 "%s" $obj

PHP代码 

class Template {
    public $content;
    
    public function __toString() {
        echo "__toString被调用\n";
        return "
" . $this->content . "
";
    }
}

$tpl = new Template();
$tpl->content = "Hello World";

// 多种触发方式
echo $tpl;           // 1. echo
print $tpl;          // 2. print
var_dump("$tpl");  // 3. 字符串中插值
运行结果 
__toString被调用

Hello World
__toString被调用

Hello World
__toString被调用
string(15) "
Hello World
"

方法调用方法

__call($name, $arguments)

调用不可访问的实例方法时触发

当调用一个不存在的实例方法或private/public方法时自动调用。

PHP代码 

class API {
    public function __call($method, $args) {
        echo "调用方法: $method\n";
        echo "参数: " . implode(", ", $args) . "\n";
        return "方法不存在，已拦截";
    }
}

$api = new API();
echo $api->getUser(1, "admin");
运行结果 
调用方法: getUser
参数: 1, admin
方法不存在，已拦截

__callStatic($name, $arguments)

调用不可访问的静态方法时触发

当调用一个不存在的静态方法时自动调用。

PHP代码 

class Math {
    public static function __callStatic($name, $args) {
        echo "静态调用: $name\n";
        return "已拦截并处理";
    }
}

echo Math::factorial(5);
运行结果 
静态调用: factorial
已拦截并处理

__invoke($arguments)

将对象当作方法调用时触发

当使用$obj()方式调用对象时自动调用，类似将对象作为函数使用。

PHP代码 

class Multiplier {
    public $$factor;
    
    public function __construct($factor) {
        $this->factor = $factor;
    }
    
    public function __invoke($value) {
        echo "invoke被调用，运算: " . $value * $this->factor . "\n";
        return $value * $this->factor;
    }
}

$double = new Multiplier(2);
$result = $double(10);
echo "结果: $result\n";
运行结果 
invoke被调用，运算: 20
结果: 20

序列化相关方法

__sleep() / __serialize()

执行serialize()时触发

控制序列化的属性。__serialize()优先级高于__sleep()。

PHP代码 

class User {
    public $name;
    public $password;
    public $token;
    
    public function __serialize() {
        echo "__serialize被调用\n";
        return [
            "name" => $this->name,
            "password" => $this->password
        ];
    }
}

$user = new User();
$user->name = "alice";
$user->password = "secret";
$user->token = "abc123";

$data = serialize($user);
echo $data;
运行结果 
__serialize被调用
O:4:"User":2:{s:4:"name";s:5:"alice";s:8:"password";s:5:"secret";}
// 注意：token被忽略了

__wakeup() / __unserialize($data)

执行unserialize()时触发

定制反序列化的逻辑。__unserialize()优先级高于__wakeup()。__wakeup()常用于绕过！

PHP代码 

class Connection {
    public $dbname;
    public $host;
    
    public function __wakeup() {
        echo "__wakeup被调用\n";
        $this->host = "localhost";
    }
}

$data = 'O:10:"Connection":2:{s:5:"dbname";s:3:"app";s:4:"host";s:9:"remote";}';
$conn = unserialize($data);
echo "Host: " . $conn->host . "\n";
运行结果 
__wakeup被调用
Host: localhost
// __wakeup将host重置为localhost

其他方法

__clone()

对象被clone时触发

当使用clone关键字复制对象时触发。

class Proto {
    public function __clone() {
        echo "对象被复制\n";
    }
}
$a = new Proto();
$b = clone $a;

__debugInfo()

var_dump/print_r时触发

当使用var_dump()或print_r()时触发。

class Hidden {
    public function __debugInfo() {
        return ["visible" => true];
    }
}

魔术方法速查表

方法 触发条件 核心作用
__construct() 对象被创建 初始化属性
__destruct() 对象被销毁 销毁前操作，漏洞入口
__get() 读取不可访问属性 属性访问控制，POP链常用
__set() 修改不可访问属性 属性赋值控制
__isset() isset()/empty()不可访问属性 检查属性存在性
__unset() unset()不可访问属性 销毁属性
__toString() 对象当字符串使用 字符串转换，漏洞利用核心
__call() 调用不可访问的实例方法 方法调用控制，SSRF利用
__callStatic() 调用不可访问的静态方法 静态方法调用控制
__invoke() 将对象当作方法调用 可调用对象
__sleep() 执行serialize() 定制序列化属性
__wakeup() 执行unserialize() 定制反序列化，绕过常在此
__clone() 对象被clone() 定制复制逻辑
__debugInfo() var_dump()/print_r() 定制调试输出
